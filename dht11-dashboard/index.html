    const ctx = document.getElementById('chart').getContext('2d');

    const chart = new Chart(ctx, {

      type: 'line',

      data: {

        labels: [],

        datasets: [

          {

            label: 'Température (°C)',

            borderColor: 'red',

            backgroundColor: 'rgba(255,0,0,0.1)',

            data: [],

            yAxisID: 'y1'

          },

          {

            label: 'Humidité (%)',

            borderColor: 'blue',

            backgroundColor: 'rgba(0,0,255,0.1)',

            data: [],

            yAxisID: 'y2'

          }

        ]

      },

      options: {

        responsive: true,

        maintainAspectRatio: false,

        scales: {

          y1: {

            beginAtZero: true,

            position: 'left',

            title: {

              display: true,

              text: 'Température (°C)'

            }

          },

          y2: {

            beginAtZero: true,

            position: 'right',

            title: {

              display: true,

              text: 'Humidité (%)'

            },

            grid: {

              drawOnChartArea: false

            }

          }

        }

      }

    });


    async function updateChart() {

     try {

       const response = await fetch('data.json', { cache: 'no-store' });

       const data = await response.json();


       // Debug brut

       document.getElementById('raw').textContent = JSON.stringify(data, null, 2);


       // Validation du JSON

       if (!data.timestamp || isNaN(data.temperature) || isNaN(data.humidity)) {

         console.warn("⛔ Données invalides ou absentes dans data.json");


         // Affiche un message clair dans le graphique

         document.getElementById('chart').style.display = 'none';

         document.getElementById('raw').textContent = '⛔ Aucune donnée valide disponible. Vérifiez votre capteur.';

         return;

       }


       // Sinon, traitement normal

       document.getElementById('chart').style.display = 'block';

       const time = new Date(data.timestamp).toLocaleTimeString();

       chart.data.labels.push(time);

       chart.data.datasets[0].data.push(data.temperature);

       chart.data.datasets[1].data.push(data.humidity);


       if (chart.data.labels.length > 30) {

         chart.data.labels.shift();

         chart.data.datasets[0].data.shift();

         chart.data.datasets[1].data.shift();

       }


       chart.update();

     } catch (err) {

       console.error("⛔ Erreur lors du chargement ou parsing du JSON", err);

       document.getElementById('chart').style.display = 'none';

       document.getElementById('raw').textContent = '⛔ Erreur de connexion ou de lecture de data.json.';

     }

   }


   updateChart();

   setInterval(updateChart, 5000);
